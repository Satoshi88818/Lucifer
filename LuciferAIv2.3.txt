Falls back gracefully if the socket path is unavailable. """PySR symbolic regression — multi-objective Pareto. Unchanged from v2.2.""" import numpy as np, pandas as pd from pysr import PySRRegressor def compute_physical_interpretability(eq_str, feature_cols): score = 0.5 for p in ['sin','cos','exp','log','sqrt','pow']: if p in eq_str: score = min(score + 0.05, 1.0) score = max(score - 0.02 * max(0, eq_str.count('(') - 3), 0.0) return score def conservation_compliance_score(eq_str, X, t_col_idx=None): if t_col_idx is not None and f'x{t_col_idx}' not in eq_str: return 1.0 return 0.7 def compute_pareto_ranks(minimize_matrix, maximize_matrix): combined = np.hstack([minimize_matrix, -maximize_matrix]) n = len(combined); ranks = np.zeros(n, dtype=int) remaining = set(range(n)); rank = 1 while remaining: front = [i for i in remaining if not any( i != j and all(combined[j] <= combined[i]) and any(combined[j] < combined[i]) for j in remaining)] for i in front: ranks[i] = rank remaining -= set(front); rank += 1 return ranks def discover_law_pareto(csv_path, target_col, feature_cols, n_iterations=200, parsimony=5e-4): df = pd.read_csv(csv_path); X = df[feature_cols].values; y = df[target_col].values model = PySRRegressor(niterations=n_iterations, binary_operators=["+","-","*","/","^"], unary_operators=["exp","log","sqrt","sin","cos","abs"], parsimony=parsimony, populations=30, maxsize=30, verbosity=0) model.fit(X, y) pareto_data = [] for _, row in model.equations_.iterrows(): eq_str = str(row['sympy_format']) pareto_data.append({ 'complexity': row['complexity'], 'mse': row['loss'], 'physical_interpretability': compute_physical_interpretability(eq_str, feature_cols), 'conservation_compliance': conservation_compliance_score(eq_str, X), 'equation': eq_str}) df_p = pd.DataFrame(pareto_data) df_p['pareto_rank'] = compute_pareto_ranks( df_p[['complexity','mse']].values, df_p[['physical_interpretability','conservation_compliance']].values) df_p = df_p.sort_values('pareto_rank'); best = df_p.iloc[0] ss_res = np.sum((y - model.predict(X))**2) ss_tot = np.sum((y - np.mean(y))**2) print(f"SR_DISCOVERED_EQUATION: {best['equation']}") print(f"SR_COMPLEXITY: {best['complexity']}") print(f"SR_R2: {1 - ss_res/(ss_tot+1e-30):.6f}") print("SR_PARETO_FRONT:") for _, row in df_p.iterrows(): print(f" rank={row['pareto_rank']} complexity={row['complexity']} " f"mse={row['mse']:.3e} phys={row['physical_interpretability']:.2f} " f"eq={row['equation']}") return model, df_p.to_dict('records')

v2.3 Design Principles

Grok is a first-class peer, not a bolt-on. It drives the Supervisor, MetaLearner, LeadResearcher, and IntuitionAgent roles by default in config.toml, and it generates the initial hypothesis in HybridGrokLucifer mode.

Routing is physics-aware, not just confidence-based. The force_pipeline_patterns config list ensures that deep derivation keywords (lagrangian, navier-stokes, etc.) always invoke the full Lucifer pipeline regardless of confidence score.

The boundary is bidirectional. Grok can call Lucifer as an HTTP tool for any deep physics/math problem, and Lucifer exports its hard-won ScientificLesson guardrails as JSONL training data that improves future Grok models.

Failures are currency. Every failed solve that generates a ScientificLesson with success_count >= 3 is automatically exported to the fine-tuning pipeline — turning Lucifer's mistakes into Grok's knowledge.

The Prompt is still alive. Grok-facing prompts (grok_hypothesis.prompt, hybrid_verifier.prompt) follow the same evolutionary injection pattern as all other Lucifer prompts — domain-relevant guardrails are dynamically prepended at call time.
