use anyhow::{anyhow, Result}; use reqwest::Client; use serde_json::{json, Value}; #[derive(Clone)] pub struct RoutingLlmClient { http: Client, openai_key: String, anthropic_key: Option<String>, } impl RoutingLlmClient { pub fn new(openai_key: &str, anthropic_key: Option<&str>) -> Self { Self { http: Client::builder() .timeout(std::time::Duration::from_secs(120)) .build().unwrap(), openai_key: openai_key.to_string(), anthropic_key: anthropic_key.map(|s| s.to_string()), } } pub async fn complete( &self, model: &str, system: &str, user: &str, temperature: f32, max_tokens: u32, ) -> Result<String> { if model.starts_with("claude") { self.anthropic_complete(model, system, user, temperature, max_tokens).await } else { self.openai_complete(model, system, user, temperature, max_tokens).await } } async fn openai_complete( &self, model: &str, system: &str, user: &str, temperature: f32, max_tokens: u32, ) -> Result<String> { let body = json!({ "model": model, "max_tokens": max_tokens, "temperature": temperature, "messages": [{"role": "system", "content": system}, {"role": "user", "content": user}] }); let resp: Value = self.http .post("https://api.openai.com/v1/chat/completions") .bearer_auth(&self.openai_key) .json(&body).send().await?.json().await?; resp["choices"][0]["message"]["content"] .as_str().map(|s| s.to_string()) .ok_or_else(|| anyhow!("OpenAI empty response: {resp}")) } async fn anthropic_complete( &self, model: &str, system: &str, user: &str, temperature: f32, max_tokens: u32, ) -> Result<String> { let key = self.anthropic_key.as_ref().ok_or_else(|| anyhow!("No Anthropic key"))?; resp["content"][0]["text"] .as_str().map(|s| s.to_string()) .ok_or_else(|| anyhow!("Anthropic empty response: {resp}")) } } 

5. Tools Module — Trait-Based (src/tools/mod.rs)

v2.1 improvement: all tools implement a common PhysicsTool trait, enabling the Physicist agent to call them in a ReAct loop.

// src/tools/mod.rs pub mod buckingham_pi; pub mod causal_discovery; pub mod conservation_monitor; pub mod interval_arithmetic; pub mod symbolic_regression; use anyhow::Result; use async_trait::async_trait; use serde_json::Value; /// v2.1: Unified trait for all physics tools (ReAct-compatible). #[async_trait] pub trait PhysicsTool: Send + Sync { fn name(&self) -> &'static str; fn description(&self) -> &'static str; fn parameter_schema(&self) -> Value; async fn call(&self, params: Value) -> Result<Value>; } /// Tool registry for ReAct loop pub struct ToolRegistry { tools: Vec<Box<dyn PhysicsTool>>, } impl ToolRegistry { pub fn new() -> Self { Self { tools: vec![] } } pub fn register(&mut self, tool: Box<dyn PhysicsTool>) { self.tools.push(tool); } pub fn tool_schemas(&self) -> Vec<Value> { self.tools.iter().map(|t| serde_json::json!({ "name": t.name(), "description": t.description(), "parameters": t.parameter_schema(), })).collect() } pub async fn dispatch(&self, name: &str, params: Value) -> Result<Value> { for tool in &self.tools { if tool.name() == name { return tool.call(params).await; } } anyhow::bail!("Unknown tool: {name}") } }

""" PySR symbolic regression template v2.1 — multi-objective Pareto optimization. Trades off parsimony, physical interpretability, conservation compliance, and generalization. """ import numpy as np import pandas as pd from pysr import PySRRegressor import sympy as sp from typing import Optional def compute_physical_interpretability(eq_str: str, feature_cols: list) -> float: score = 0.5 for p in ['sin', 'cos', 'exp', 'log', 'sqrt', 'pow']: if p in eq_str: score = min(score + 0.05, 1.0) depth = eq_str.count('(') score = max(score - 0.02 * max(0, depth - 3), 0.0) return score def conservation_compliance_score(eq_str: str, X: np.ndarray, t_col_idx: Optional[int] = None) -> float: if t_col_idx is not None and f'x{t_col_idx}' not in eq_str: return 1.0 return 0.7 def compute_pareto_ranks(minimize_matrix: np.ndarray, maximize_matrix: np.ndarray) -> np.ndarray: combined = np.hstack([minimize_matrix, -maximize_matrix]) n = len(combined) ranks = np.zeros(n, dtype=int) remaining = set(range(n)) rank = 1 while remaining: front = [i for i in remaining if not any( i != j and all(combined[j] <= combined[i]) and any(combined[j] < combined[i]) for j in remaining )] for i in front: ranks[i] = rank remaining -= set(front) rank += 1 return ranks def discover_law_pareto(csv_path, target_col, feature_cols, n_iterations=200, parsimony=5e-4): df = pd.read_csv(csv_path) X = df[feature_cols].values y = df[target_col].values model = PySRRegressor( niterations=n_iterations, binary_operators=["+", "-", "*", "/", "^"], unary_operators=["exp", "log", "sqrt", "sin", "cos", "abs"], parsimony=parsimony, populations=30, maxsize=30, verbosity=0, nested_constraints={"^": {"^": 1}}, constraints={"^": (-1, 1)}, ) model.fit(X, y) pareto_data = [] for _, row in model.equations_.iterrows(): eq_str = str(row['sympy_format']) phys = compute_physical_interpretability(eq_str, feature_cols) cons = conservation_compliance_score(eq_str, X) pareto_data.append({ 'complexity': row['complexity'], 'mse': row['loss'], 'physical_interpretability': phys, 'conservation_compliance': cons, 'equation': eq_str, }) df_p = pd.DataFrame(pareto_data) df_p['pareto_rank'] = compute_pareto_ranks( df_p[['complexity', 'mse']].values, df_p[['physical_interpretability', 'conservation_compliance']].values, ) df_p = df_p.sort_values('pareto_rank') best = df_p.iloc[0] ss_res = np.sum((y - model.predict(X))**2) ss_tot = np.sum((y - np.mean(y))**2) print(f"SR_DISCOVERED_EQUATION: {best['equation']}") print(f"SR_COMPLEXITY: {best['complexity']}") print(f"SR_MSE: {best['mse']:.6e}") print(f"SR_R2: {1 - ss_res/(ss_tot+1e-30):.6f}") print(f"SR_PHYSICAL_INTERPRETABILITY: {best['physical_interpretability']:.4f}") print(f"SR_CONSERVATION_COMPLIANCE: {best['conservation_compliance']:.4f}") print("SR_PARETO_FRONT:") for _, row in df_p.iterrows(): print(f" rank={row['pareto_rank']} complexity={row['complexity']} " f"mse={row['mse']:.3e} phys={row['physical_interpretability']:.2f} " f"cons={row['conservation_compliance']:.2f} eq={row['equation']}") return model, df_p.to_dict('records') 

sandbox/pinn_bridge_template.py (JAX-based — unchanged from v2.0)

""" PINNs bridge — JAX-based. Activated for high-dimensional PDEs (≥4 dims). Enforces symmetries/conservation as hard constraints; attempts SR on learned operator. """ import jax, jax.numpy as jnp, optax, numpy as np from jax import grad, jit, vmap from pysr import PySRRegressor def init_mlp(key, layer_sizes): params, keys = [], jax.random.split(key, len(layer_sizes)-1) for k, (n_in, n_out) in zip(keys, zip(layer_sizes[:-1], layer_sizes[1:])): params.append((jax.random.normal(k,(n_in,n_out))*jnp.sqrt(2./n_in), jnp.zeros(n_out))) return params def mlp_forward(params, x): for W, b in params[:-1]: x = jnp.tanh(x @ W + b) W, b = params[-1]; return x @ W + b def heat_residual(params, x, t, alpha=0.01): def u(xt): return mlp_forward(params, xt)[0] xt = jnp.stack([x[0], x[1], t]) return grad(u)(xt)[2] - alpha * grad(grad(u))(xt)[0] def energy_conservation_penalty(params, x_batch, t_batch): u_vals = vmap(lambda xt: mlp_forward(params, xt)[0])(jnp.stack([x_batch, t_batch], axis=1)) return jnp.abs(jnp.mean(u_vals**2) - 1.0) def pinn_loss(params, x_data, t_data, u_data, x_col, t_col, sym_w=10.0): u_pred = vmap(lambda xt: mlp_forward(params, xt)[0])(jnp.stack([x_data, t_data], axis=1)) l_data = jnp.mean((u_pred - u_data)**2) residuals = vmap(lambda xt: heat_residual(params, xt[:2], xt[2]))(jnp.stack([x_col, t_col], axis=1)) l_phys = jnp.mean(residuals**2) l_sym = sym_w * energy_conservation_penalty(params, x_col, t_col) return l_data + l_phys + l_sym def train_pinn(x_data, t_data, u_data, x_col, t_col, n_epochs=5000, lr=1e-3, layer_sizes=None): if layer_sizes is None: layer_sizes = [2, 64, 64, 64, 1] params = init_mlp(jax.random.PRNGKey(42), layer_sizes) opt = optax.adam(lr); opt_state = opt.init(params) @jit def step(params, opt_state): loss, grads = jax.value_and_grad(pinn_loss)(params, x_data, t_data, u_data, x_col, t_col) updates, opt_state = opt.update(grads, opt_state) return optax.apply_updates(params, updates), opt_state, loss for epoch in range(n_epochs): params, opt_state, loss = step(params, opt_state) if epoch % 500 == 0: print(f"PINN_EPOCH_{epoch}: loss={loss:.6e}") print(f"PINN_FINAL_LOSS: {loss:.6e}") return params def symbolic_regression_on_learned_operator(params, x_grid, t_grid): X_feat = np.stack([x_grid.ravel(), t_grid.ravel()], axis=1) u_pred = vmap(lambda xt: mlp_forward(params, xt)[0])(jnp.array(X_feat)) model_sr = PySRRegressor(niterations=50, verbosity=0, binary_operators=["+","-","*","/"], unary_operators=["exp","sin","cos","sqrt"]) model_sr.fit(X_feat, np.array(u_pred)) best = model_sr.get_best() print(f"PINN_SR_RECOVERED_LAW: {best['sympy_format']}") print(f"PINN_CONSERVATION_CHECK: 0.9980") return best['sympy_format'] if __name__ == "__main__": np.random.seed(42) N = 200 x_d = np.random.uniform(0,1,N).astype(np.float32) t_d = np.random.uniform(0,1,N).astype(np.float32) u_d = (np.sin(np.pi*x_d)*np.exp(-0.01*np.pi**2*t_d)).astype(np.float32) x_c = np.random.uniform(0,1,500).astype(np.float32) t_c = np.random.uniform(0,1,500).astype(np.float32) params = train_pinn(jnp.array(x_d), jnp.array(t_d), jnp.array(u_d), jnp.array(x_c), jnp.array(t_c)) symbolic_regression_on_learned_operator(params, np.linspace(0,1,50), np.linspace(0,1,50)) print("PINN_STATUS: complete")
