use anyhow::{anyhow, Result}; use reqwest::Client; use serde_json::{json, Value}; #[derive(Clone)] pub struct RoutingLlmClient { http: Client, openai_key: String, anthropic_key: Option<String>, } impl RoutingLlmClient { pub fn new(openai_key: &str, anthropic_key: Option<&str>) -> Self { Self { http: Client::builder() .timeout(std::time::Duration::from_secs(120)) .build().unwrap(), openai_key: openai_key.to_string(), anthropic_key: anthropic_key.map(|s| s.to_string()), } } pub async fn complete( &self, model: &str, system: &str, user: &str, temperature: f32, max_tokens: u32, ) -> Result<String> { if model.starts_with("claude") { self.anthropic_complete(model, system, user, temperature, max_tokens).await } else { self.openai_complete(model, system, user, temperature, max_tokens).await } } async fn openai_complete( &self, model: &str, system: &str, user: &str, temperature: f32, max_tokens: u32, ) -> Result<String> { let body = json!({ "model": model, "max_tokens": max_tokens, "temperature": temperature, "messages": [ {"role": "system", "content": system}, {"role": "user", "content": user} ] }); let resp: Value = self.http.post("https://api.openai.com/v1/chat/completions") .bearer_auth(&self.openai_key) .json(&body).send().await?.json().await?; resp["choices"][0]["message"]["content"] .as_str().map(|s| s.to_string()) .ok_or_else(|| anyhow!("OpenAI empty response: {resp}")) } async fn anthropic_complete( &self, model: &str, system: &str, user: &str, temperature: f32, max_tokens: u32, ) -> Result<String> { let key = self.anthropic_key.as_ref().ok_or_else(|| anyhow!("No Anthropic key"))?; let body = json!({ "model": model, "max_tokens": max_tokens, "temperature": temperature, "system": system, "messages": [{"role": "user", "content": user}] }); let resp: Value = self.http.post("https://api.anthropic.com/v1/messages") .header("x-api-key", key) .header("anthropic-version", "2023-06-01") .json(&body).send().await?.json().await?; resp["content"][0]["text"] .as_str().map(|s| s.to_string()) .ok_or_else(|| anyhow!("Anthropic empty response: {resp}")) } }

"""SALib Sobol sensitivity analysis template.""" import numpy as np from SALib.sample import saltelli from SALib.analyze import sobol def run_sensitivity(model_fn, param_bounds: dict, n_samples: int = 512): names = list(param_bounds.keys()) bounds = list(param_bounds.values()) problem = {"num_vars": len(names), "names": names, "bounds": bounds} X = saltelli.sample(problem, n_samples, calc_second_order=False) Y = np.array([model_fn(x) for x in X]) Si = sobol.analyze(problem, Y, print_to_console=False) print("SOBOL_SENSITIVITY:") for name, s1, st in zip(names, Si["S1"], Si["ST"]): print(f" {name}: S1={s1:.4f} ST={st:.4f}") print(f"MOST_SENSITIVE_PARAM: {names[np.argmax(Si['ST'])]}") return Si
