use pest_derive::Parser; use anyhow::{bail, Result}; use crate::types::Expr; /// The PEG grammar is embedded via pest_derive. /// The grammar file is at `src/symbolic/physics.pest`. #[derive(Parser)] #[grammar = "symbolic/physics.pest"] pub struct PhysicsParser; /// // Returns: Sub(Mul(Mul(0.5, m), Pow(v, 2)), Mul(Mul(0.5, k), Pow(x, 2))) /// ``` pub fn parse(&self, input: &str) -> Result<Expr> { let pairs = PhysicsParser::parse(Rule::expr, input.trim()) .map_err(|e| anyhow::anyhow!("PEG parse error: {}", e))?; Ok(Expr::Number(v)) } Rule::symbol => Ok(Expr::Symbol(pair.as_str().to_string())), Rule::func_call => self.parse_func_call(pair), _ => bail!("Unexpected rule: {:?}", pair.as_rule()), } } fn parse_additive(&self, mut pairs: pest::iterators::Pairs<Rule>) -> Result<Expr> { let mut left = self.build_expr(pairs.next().ok_or_else(|| anyhow::anyhow!("Missing LHS"))?)?; while let Some(op_pair) = pairs.next() { let op = op_pair.as_str(); let right = self.build_expr(pairs.next().ok_or_else(|| anyhow::anyhow!("Missing RHS"))?)?; left = match op { "+" => Expr::Add(Box::new(left), Box::new(right)), "-" => Expr::Sub(Box::new(left), Box::new(right)), _ => bail!("Unknown additive op: {op}"), }; } Ok(left) } fn parse_term(&self, mut pairs: pest::iterators::Pairs<Rule>) -> Result<Expr> { let mut left = self.build_expr(pairs.next().ok_or_else(|| anyhow::anyhow!("Missing LHS"))?)?; while let Some(op_pair) = pairs.next() { let op = op_pair.as_str(); let right = self.build_expr(pairs.next().ok_or_else(|| anyhow::anyhow!("Missing RHS"))?)?; left = match op { "*" => Expr::Mul(Box::new(left), Box::new(right)), "/" => Expr::Div(Box::new(left), Box::new(right)), _ => bail!("Unknown multiplicative op: {op}"), }; } Ok(left) } fn parse_power(&self, mut pairs: pest::iterators::Pairs<Rule>) -> Result<Expr> { let base = self.build_expr(pairs.next().ok_or_else(|| anyhow::anyhow!("Missing base"))?)?; if let Some(exp) = pairs.next() { let exponent = self.build_expr(exp)?; Ok(Expr::Pow(Box::new(base), Box::new(exponent))) } else { Ok(base) } } fn parse_factor(&self, mut pairs: pest::iterators::Pairs<Rule>) -> Result<Expr> { let first = pairs.next().ok_or_else(|| anyhow::anyhow!("Empty factor"))?; if first.as_str() == "-" { let inner = self.build_expr(pairs.next().ok_or_else(|| anyhow::anyhow!("Missing neg operand"))?)?; Ok(Expr::Neg(Box::new(inner))) } else { self.build_expr(first) } } fn parse_atom(&self, pair: pest::iterators::Pair<Rule>) -> Result<Expr> { let inner = pair.into_inner().next() .ok_or_else(|| anyhow::anyhow!("Empty atom"))?; self.build_expr(inner) } fn parse_func_call(&self, pair: pest::iterators::Pair<Rule>) -> Result<Expr> { let mut inner = pair.into_inner(); let func_name = inner.next() .ok_or_else(|| anyhow::anyhow!("Missing function name"))?.as_str().to_string(); let args: Result<Vec<Expr>> = inner.map(|p| self.build_expr(p)).collect(); let args = args?; // Handle special calculus operators match func_name.as_str() { "diff" | "d" => { if args.len() >= 2 { if let Expr::Symbol(var) = &args[1] { return Ok(Expr::Derivative { body: Box::new(args[0].clone()), var: var.clone(), }); } } Ok(Expr::Apply { func: func_name, args }) } "integral" | "int" => { if args.len() >= 2 { if let Expr::Symbol(var) = &args[1] { return Ok(Expr::Integral { body: Box::new(args[0].clone()), var: var.clone(), }); } } Ok(Expr::Apply { func: func_name, args }) } _ => Ok(Expr::Apply { func: func_name, args }), } } }

// physics.pest — PEG grammar for physics mathematical expressions // Handles: arithmetic, exponents, functions, Greek symbols, calculus notation WHITESPACE = _{ " " | "\t" | "\n" | "\r" } // Top-level expression expr = { additive } additive = { term ~ (add_op ~ term)* } add_op = _{ "+" | "-" } term = { factor ~ (mul_op ~ factor)* } mul_op = _{ "*" | "/" } factor = { "-" ~ power | power } power = { atom ~ ("^" ~ atom)? } atom = { "(" ~ expr ~ ")" | func_call | number | symbol } // Function calls: sin(x), diff(f, x), integral(f, x, 0, inf) func_call = { func_name ~ "(" ~ expr ~ ("," ~ expr)* ~ ")" } func_name = @{ ASCII_ALPHA ~ (ASCII_ALPHANUMERIC | "_")* } // Numbers: integer, decimal, scientific notation number = @{ "-"? ~ ASCII_DIGIT+ ~ ("." ~ ASCII_DIGIT+)? ~ (("e" | "E") ~ ("+"|"-")? ~ ASCII_DIGIT+)? } // Symbols: alphanumeric + underscore + selected Greek Unicode characters symbol = @{ (ASCII_ALPHA | greek_letter | "∂" | "∇") ~ (ASCII_ALPHANUMERIC | "_" | greek_letter | "'" | "·")* } greek_letter = _{ "α" | "β" | "γ" | "Γ" | "δ" | "Δ" | "ε" | "ζ" | "η" | "θ" | "Θ" | "ι" | "κ" | "λ" | "Λ" | "μ" | "ν" | "ξ" | "Ξ" | "π" | "Π" | "ρ" | "σ" | "Σ" | "τ" | "υ" | "φ" | "Φ" | "χ" | "ψ" | "Ψ" | "ω" | "Ω" | "ℏ" }

src/persistence/distributed_cache.rs (NEW v2.5)

//! v2.5: Distributed proof + tactic cache. //! //! Replaces SQLite-only storage for the Lean 4 tactic cache. //! Cache lookup order: Redis (hot) → PostgreSQL (warm) → SQLite (local fallback). //! //! Multiple Lucifer instances can share learned lemmas globally via //! the shared PostgreSQL + Redis layer — "Collective Scientific Memory". use anyhow::Result; use redis::AsyncCommands; use tokio_postgres::Client as PgClient; pub struct DistributedProofCache { redis: redis::aio::ConnectionManager, postgres: PgClient, } impl DistributedProofCache { pub async fn new(redis_url: &str, postgres_url: &str) -> Result<Self> { // Redis connection let redis_client = redis::Client::open(redis_url)?; let redis = redis::aio::ConnectionManager::new(redis_client).await?; // PostgreSQL connection let (pg_client, pg_conn) = tokio_postgres::connect(postgres_url, tokio_postgres::NoTls).await?; tokio::spawn(async move { if let Err(e) = pg_conn.await { tracing::error!("PG connection error: {e}"); } }); // Ensure schema exists pg_client.execute( "CREATE TABLE IF NOT EXISTS tactic_cache ( stub_hash TEXT PRIMARY KEY, tactic_seq TEXT NOT NULL, success BOOLEAN NOT NULL, lean4_proof TEXT, domain TEXT, used_count INTEGER DEFAULT 1, created_at TIMESTAMPTZ DEFAULT NOW(), last_used TIMESTAMPTZ DEFAULT NOW() )", &[] ).await?; Ok(Self { redis, postgres: pg_client }) } /// Look up a tactic sequence for a Lean 4 stub hash. /// Hot path: Redis. Warm path: PostgreSQL. pub async fn get_tactics(&mut self, stub_hash: &str) -> Result<Option<String>> { // Try Redis first (sub-millisecond) let redis_result: Option<String> = self.redis.get( format!("tactic:{stub_hash}") ).await.unwrap_or(None); if let Some(tactics) = redis_result { tracing::debug!("🔥 Redis tactic cache hit: {stub_hash}"); return Ok(Some(tactics)); } // Fall back to PostgreSQL let row = self.postgres.query_opt( "SELECT tactic_seq FROM tactic_cache WHERE stub_hash = $1 AND success = true", &[&stub_hash] ).await?; if let Some(r) = row { let tactics: String = r.get(0); tracing::debug!("🐘 PostgreSQL tactic cache hit: {stub_hash}"); // Promote to Redis for next time (TTL: 1 hour) let _: () = self.redis.set_ex( format!("tactic:{stub_hash}"), &tactics, 3600 ).await.unwrap_or(()); return Ok(Some(tactics)); } Ok(None) } /// Store a successful tactic sequence in the distributed cache. pub async fn store_tactics( &mut self, stub_hash: &str, tactic_seq: &str, lean4_proof: Option<&str>, domain: Option<&str>, ) -> Result<()> { // Write to Redis (TTL: 1 hour for hot cache) let _: () = self.redis.set_ex( format!("tactic:{stub_hash}"), tactic_seq, 3600 ).await.unwrap_or(()); // Write to PostgreSQL (permanent) self.postgres.execute( "INSERT INTO tactic_cache (stub_hash, tactic_seq, success, lean4_proof, domain) VALUES ($1, $2, true, $3, $4) ON CONFLICT (stub_hash) DO UPDATE SET used_count = tactic_cache.used_count + 1, last_used = NOW()", &[&stub_hash, &tactic_seq, &lean4_proof, &domain] ).await?; tracing::info!("💾 Tactic sequence cached globally: {stub_hash}"); Ok(()) } /// Get globally learned lemmas for a physics domain. /// Used by IterativeProverAgent for warm-start tactic selection. pub async fn get_top_lemmas_for_domain(&self, domain: &str, limit: i64) -> Result<Vec<String>> { let rows = self.postgres.query( "SELECT tactic_seq FROM tactic_cache WHERE domain = $1 AND success = true ORDER BY used_count DESC LIMIT $2", &[&domain, &limit] ).await?; Ok(rows.iter().map(|r| r.get::<_, String>(0)).collect()) } }

v2.5 Design Principles

The parser is the foundation. Every downstream improvement — dimensional inference, E-Graph saturation, Lie algebra checks, Lean 4 tactic generation — is only as reliable as the parser that produces the AST. The PEG grammar is now the bedrock of the entire symbolic pipeline.

Security is structural, not configurational. gVisor and Firecracker prevent sandbox escapes at the kernel/hypervisor level. Zero-Trust networking means the sandbox cannot exfiltrate API keys regardless of what the LLM-generated code attempts. The threat model is enforced by hardware, not config flags.

The lab scales horizontally. GPU-intensive PINNs and CPU-intensive Lean 4 MCTS are now K8s Jobs. The orchestrator submits work and waits — it is no longer a scaling bottleneck. The proof cache is shared globally across all Lucifer instances via Redis + PostgreSQL.

Every discovered law has a provenance chain. MLflow tracks every symbolic regression result. The Triple-Lock mechanism ensures only equations that survive symbolic, simulation, and formal proof validation are promoted to ground truth in the causal graph. Human physicists are the final gatekeepers via the Scientific Git PR workflow.

The system improves through rejection. When a human physicist rejects a law PR, the rejection reason becomes a ScientificLesson fed to the MetaLearnerAgent — closing the loop between human expertise and automated discovery.
