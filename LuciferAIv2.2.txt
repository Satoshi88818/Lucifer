// Unchanged from v2.1 — routing between OpenAI and Anthropic APIs. use anyhow::{anyhow, Result}; use reqwest::Client; use serde_json::{json, Value}; #[derive(Clone)] pub struct RoutingLlmClient { http: Client, openai_key: String, anthropic_key: Option<String>, } impl RoutingLlmClient { pub fn new(openai_key: &str, anthropic_key: Option<&str>) -> Self { Self { http: Client::builder() .timeout(std::time::Duration::from_secs(120)) .build().unwrap(), openai_key: openai_key.to_string(), anthropic_key: anthropic_key.map(|s| s.to_string()), } } pub async fn complete( &self, model: &str, system: &str, user: &str, temperature: f32, max_tokens: u32, ) -> Result<String> { if model.starts_with("claude") { self.anthropic_complete(model, system, user, temperature, max_tokens).await } else { self.openai_complete(model, system, user, temperature, max_tokens).await } } async fn openai_complete( &self, model: &str, system: &str, user: &str, temperature: f32, max_tokens: u32, ) -> Result<String> { let body = json!({ "model": model, "max_tokens": max_tokens, "temperature": temperature, "messages": [{"role":"system","content":system}, {"role":"user","content":user}] }); let resp: Value = self.http.post("https://api.openai.com/v1/chat/completions") .bearer_auth(&self.openai_key).json(&body).send().await?.json().await?; resp["choices"][0]["message"]["content"].as_str().map(|s| s.to_string()) .ok_or_else(|| anyhow!("OpenAI empty response: {resp}")) } async fn anthropic_complete( &self, model: &str, system: &str, user: &str, temperature: f32, max_tokens: u32, ) -> Result<String> { let key = self.anthropic_key.as_ref().ok_or_else(|| anyhow!("No Anthropic key"))?; let body = json!({ "model": model, "max_tokens": max_tokens, "temperature": temperature, "system": system, "messages": [{"role":"user","content":user}] }); let resp: Value = self.http.post("https://api.anthropic.com/v1/messages") .header("x-api-key", key).header("anthropic-version","2023-06-01") .json(&body).send().await?.json().await?; """ Conservation monitor v2.2 — streams energy-drift readings over a Unix socket. The Rust LiveConservationMonitor listens and kills the container if drift > 1%. Falls back gracefully if the socket path is unavailable. """ import os, socket, numpy as np from scipy.integrate import solve_ivp SOCKET_PATH = os.environ.get("LUCIFER_MONITOR_SOCKET", "/tmp/lucifer_monitor.sock") def energy(y): return 0.5 * y[1] ** 2 + 0.5 * y[0] ** 2 def ode(t, y): return [y[1], -y[0]] # Connect to monitor socket (non-blocking; fail silently) monitor_sock = None try: monitor_sock = socket.socket(socket.AF_UNIX, socket.SOCK_STREAM) monitor_sock.connect(SOCKET_PATH) except Exception: monitor_sock = None _E0 = None _max_drift = 0.0 def energy_event(t, y): global _E0, _max_drift E = energy(y) if _E0 is None: _E0 = E drift = abs(E - _E0) / (abs(_E0) + 1e-30) _max_drift = max(_max_drift, drift) # Stream drift to Rust monitor if monitor_sock: try: monitor_sock.sendall(f"{drift:.8f}".encode()) except Exception: pass return 1.0 # non-terminal event energy_event.terminal = False energy_event.direction = 0 sol = solve_ivp(ode, [0, 100], [1.0, 0.0], max_step=0.01, events=energy_event) if monitor_sock: try: monitor_sock.close() except Exception: pass print(f"CONSERVATION_ENERGY_DRIFT: {_max_drift:.2e}")

"""PySR symbolic regression — multi-objective Pareto. Unchanged from v2.1.""" import numpy as np, pandas as pd from pysr import PySRRegressor def compute_physical_interpretability(eq_str, feature_cols): score = 0.5 for p in ['sin','cos','exp','log','sqrt','pow']: if p in eq_str: score = min(score + 0.05, 1.0) score = max(score - 0.02 * max(0, eq_str.count('(') - 3), 0.0) return score def conservation_compliance_score(eq_str, X, t_col_idx=None): if t_col_idx is not None and f'x{t_col_idx}' not in eq_str: return 1.0 return 0.7 def compute_pareto_ranks(minimize_matrix, maximize_matrix): combined = np.hstack([minimize_matrix, -maximize_matrix]) n = len(combined); ranks = np.zeros(n, dtype=int); remaining = set(range(n)); rank = 1 while remaining: front = [i for i in remaining if not any( i != j and all(combined[j] <= combined[i]) and any(combined[j] < combined[i]) for j in remaining)] for i in front: ranks[i] = rank remaining -= set(front); rank += 1 return ranks def discover_law_pareto(csv_path, target_col, feature_cols, n_iterations=200, parsimony=5e-4): df = pd.read_csv(csv_path); X = df[feature_cols].values; y = df[target_col].values model = PySRRegressor(niterations=n_iterations, binary_operators=["+","-","*","/","^"], unary_operators=["exp","log","sqrt","sin","cos","abs"], parsimony=parsimony, populations=30, maxsize=30, verbosity=0) model.fit(X, y) pareto_data = [] for _, row in model.equations_.iterrows(): eq_str = str(row['sympy_format']) pareto_data.append({'complexity': row['complexity'], 'mse': row['loss'], 'physical_interpretability': compute_physical_interpretability(eq_str, feature_cols), 'conservation_compliance': conservation_compliance_score(eq_str, X), 'equation': eq_str}) df_p = pd.DataFrame(pareto_data) df_p['pareto_rank'] = compute_pareto_ranks( df_p[['complexity','mse']].values, df_p[['physical_interpretability','conservation_compliance']].values) df_p = df_p.sort_values('pareto_rank'); v2.2 Design Principles

Failures are indexed data points — every failed solve() feeds MetaLearnerAgent and enriches the ScientificLesson store.

The Prompt is alive — supervisor and physicist prompts dynamically include domain-relevant guardrails ranked by evolutionary success count.

The Sandbox is efficient — warm containers eliminate cold-start latency; the live monitor reclaims GPU/CPU the microsecond conservation laws break.

Knowledge is Collective — tactic proofs and failure lessons are shared across all sessions via SQLite; LanceDB clusters failures semantically.

Resources are spent wisely — literature ingestion fires only when the system genuinely lacks knowledge confidence, not on a fixed cron schedule.
